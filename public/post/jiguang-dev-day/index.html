JG stock price with plotly:

library(quantmod)
library(plotly)

getSymbols("JG")
## [1] "JG"
JG_data=as.data.frame(JG)
JG_data$time <- rownames(JG_data)
JG_data=JG_data%>%filter(time>='2018-06-01')
p <- plot_ly(JG_data, x = ~time, y = ~JG.Close, type = 'scatter', mode = 'lines+markers')
p
