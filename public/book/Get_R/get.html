Chapter 2 Get

In this chapter,will share how to connect(both input and output) with variety of data source into R system,for example csv,excel,database ect.

2.1 Connect with csv/txt

Input csv: read a csv using [readr::read_csv].Its 8 times faster than read.csv

library(readr)
csv_data <- read_csv("zipcode.csv", header=TRUE)

Output csv: write a csv using [readr::write_csv].Its twice as fast as write.csv

library(readr)
write_csv(csv_data, "zipcode.csv", append = FALSE)

Input txt:

txt_data <- read.table("txt_data.txt", header = TRUE)

Output txt:

write.table(data001, file = "mtcars.txt", sep = "\t")

2.3 Connect with rds/RData/feather

rds:

# input a dataframe
rdsdata=readRDS(file = "my_data.rds")

#input RData:

save(data1, data2, file = "data.RData")

# input feather

# output feather
write_feather(data,'data.feather')

2.5.4 Teradata with ebaytd package

library(ebaytd)
teradataInit("YOUR_USER_NAME", "YOUR_TERADATA_PASSWORD")
# local 
conn <- teradataConnect(database= 'P_xxxx_t', system="mozart") 
connf <- teradataConnect(database='P_xxxx_t', fastload = TRUE,csv=TRUE,system="mozart")
# server 
conn <- teradataConnect(database= 'P_xxxx_t', system="mozarbe") 
connf <- teradataConnect(database='P_xxxx_t', fastload = TRUE,csv=TRUE,system="mozarbe")

#Create table at Teradata:
dbSendQuery(conn, "create multiset table P_xxxx_t.DW_COUNTRIES AS (SELECT  * from DW_COUNTRIES) WITH    DATA ")

#Download table from Teradata:  
data001=dbGetQuery(conn, " select * from P_xxxx_t.DW_COUNTRIES")


#Download table from Teradata by batch:  

res=dbsendQuery(conn, " select * from P_xxxx_t.DW_COUNTRIES")
result<-list()
e=1
result[[e]]<-dbFetch(res,n=5000)
while(nrow(chunk <- dbFetch(res, n = 5000))>0){
  e<-e+1
  result[[e]]<-chunk
  print(e)
  data001<-do.call(rbind,result)
}






#Upload table to Teradata:
teradataFastloadCSV(data001, conn, connf, "P_xxxx_t", "tera_data001", replace = TRUE, primary_index = c(1))
#Check whether a table is exist on Teradata:
dbExistsTable(conn,"P_xxxx_t","test")
#Delete table on Teradata
dbExistsRemove(conn,"P_xxxx_t","tera_data001")

Get all account Rbloggers all tweets:

#Create your own appication key at https://dev.twitter.com/apps
consumer_key = "EZRy5JzOH2QQmVAe9B4j2w";
consumer_secret = "OIDC4MdfZJ82nbwpZfoUO4WOLTYjoRhpHRAWj6JMec";
#Use basic auth
secret <- jsonlite::base64_enc(paste(consumer_key, consumer_secret, sep = ":"))
req <- httr::POST("https://api.twitter.com/oauth2/token",
  httr::add_headers(
    "Authorization" = paste("Basic", gsub("\n", "", secret)),
    "Content-Type" = "application/x-www-form-urlencoded;charset=UTF-8"
  ),
  body = "grant_type=client_credentials"

token <- paste("Bearer", httr::content(req)$access_token)
#Actual API call
url <- "https://api.twitter.com/1.1/statuses/user_timeline.json?count=10&screen_name=Rbloggers"
req <- httr::GET(url, httr::add_headers(Authorization = token))
json <- httr::content(req, as = "text")
tweets <- fromJSON(json)
substring(tweets$text, 1, 100)
