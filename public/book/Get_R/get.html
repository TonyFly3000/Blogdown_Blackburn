2.3 Connect with rds/RData/feather

rds:

# input a dataframe
rdsdata=readRDS(file = "my_data.rds")

# output a dataframe
saveRDS(rdsdata, file = "my_data.rds")

RData:

# input RData
load("my_work_space.RData")

# output a dataframe into RData
save(data1, file = "data.RData")

# output multiple dataframe into RData
save(data1, data2, file = "data.RData")

# output all dataframe into RData
save.image(file = "my_work_space.RData")

feather:

# input feather
data=read_feather('data.feather')

# output feather
write_feather(data,'data.feather')

2.5.2 Teradata with ebaytd package

library(ebaytd)
teradataInit("YOUR_USER_NAME", "YOUR_TERADATA_PASSWORD")

# local 
conn <- teradataConnect(database= 'P_xxxx_t', system="mozart") 
connf <- teradataConnect(database='P_xxxx_t', fastload = TRUE,csv=TRUE,system="mozart")

# server 
conn <- teradataConnect(database= 'P_xxxx_t', system="mozarbe") 
connf <- teradataConnect(database='P_xxxx_t', fastload = TRUE,csv=TRUE,system="mozarbe")

#Create table at Teradata:
dbSendQuery(conn, "create multiset table P_xxxx_t.DW_COUNTRIES AS (SELECT  * from DW_COUNTRIES) WITH    DATA ")

#Download table from Teradata:  
data001=dbGetQuery(conn, " select * from P_xxxx_t.DW_COUNTRIES")

2.6 Connect with salesforce

library(RForcecom)
username <- "xxx@ebay.com"
password <- "xxxxxxx"

apiVersion <- "41.0"

(session <- rforcecom.login(username, password, instanceURL, apiVersion))

all_object=rforcecom.getObjectList(session)

ILM=rforcecom.retrieve(session, 'Item_Location_Mispresentation__c', c("Id","Recommended_Action__c"))


soqlQuery <- "SELECT Id,Recommended_Action FROM Item_Location_Mispresentation__c"
