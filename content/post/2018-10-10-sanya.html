library(leaflet)
library(ggmap)
## Loading required package: ggplot2
location=geocode("Sanya", source = "dsk")
## Information from URL : http://www.datasciencetoolkit.org/maps/api/geocode/json?address=Sanya&sensor=false
lon=location$lon
lat=location$lat
m <- leaflet()%>%
  addTiles() %>%  # Add default OpenStreetMap map tiles
  setView(lng=lon, lat=lat, zoom = 05) %>%
addMarkers(lng=lon, lat=lat, popup="Sanya")
