library(leaflet)
library(ggmap)
location=geocode("Kyoto", source = "dsk")
lon=location$lon
lat=location$lat
m <- leaflet()%>%
  addTiles() %>%  # Add default OpenStreetMap map tiles
  setView(lng=lon, lat=lat, zoom = 08) %>%
addMarkers(lng=lon, lat=lat, popup="Kyoto")
