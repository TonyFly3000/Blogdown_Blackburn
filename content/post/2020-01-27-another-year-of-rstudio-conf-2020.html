library(tidyquery) library(nycflights13)

query( " SELECT origin, dest, COUNT(flight) AS num_flts, round(SUM(seats)) AS num_seats, round(AVG(arr_delay)) AS avg_delay FROM flights f LEFT OUTER JOIN planes p ON f.tailnum = p.tailnum WHERE distance BETWEEN 200 AND 300 AND air_time IS NOT NULL GROUP BY origin, dest HAVING num_flts > 3000 ORDER BY num_seats DESC, avg_delay ASC LIMIT 2;" )
