Location:

library(leaflet)
library(ggmap)
location=geocode( "Hanoi" )
m <- leaflet()%>%
  addTiles() %>%  # Add default OpenStreetMap map tiles
  setView(lng=location$lon, lat=location$lat, zoom = 05) %>%
addMarkers(lng=location$lon, lat=location$lat, popup="Hanoi")
m
