Location:

library(leaflet)
library(ggmap)
## Loading required package: ggplot2
location=geocode("Kexing Science Park shenzhen china", source = "dsk")
## Information from URL : http://www.datasciencetoolkit.org/maps/api/geocode/json?address=Kexing%20Science%20Park%20shenzhen%20china&sensor=false
lon=location$lon
lat=location$lat
m <- leaflet()%>%
  addTiles() %>%  # Add default OpenStreetMap map tiles
  setView(lng=lon, lat=lat, zoom = 15) %>%
addMarkers(lng=lon, lat=lat, popup="@Kexing Science Park")
m
