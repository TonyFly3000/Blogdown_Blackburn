library(leaflet)
library(ggmap)
location=geocode("Beijing", source = "dsk")
lon=location$lon
lat=location$lat
m <- leaflet()%>%
  addTiles() %>%  # Add default OpenStreetMap map tiles
  setView(lng=lon, lat=lat, zoom = 05) %>%
addMarkers(lng=lon, lat=lat, popup="Beijing")
m
